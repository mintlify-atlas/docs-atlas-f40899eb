## Overview

The `productos.js` module handles all product-related operations including fetching, rendering, filtering, pagination, sorting, and CRUD operations. It integrates with the backend API and manages product state globally.

### obtenerProductos()

Fetches products from the API with pagination, sorting, and filtering support.

```javascript productos.js:7-51
function obtenerProductos(pagina = 0) {
  let url = `http://localhost:8080/api/producto/listar?page=${pagina}&size=20&${sortHelper.getParams()}`;

  // Se obtienen los valores de los filtros
  const nombre = document.getElementById("filtroNombre")?.value.trim();
  const marca = document.getElementById("filtroMarca")?.value.trim();
  const categoria = document.getElementById("filtroCategoria")?.value;
  const precioMin = document.getElementById("filtroPrecioMin")?.value;
  const precioMax = document.getElementById("filtroPrecioMax")?.value;

  // Se agregan los filtros a la URL si tienen valor
  if (nombre) {
    url += `&nombre=${encodeURIComponent(nombre)}`;
  }
  if (marca) {
    url += `&marca=${encodeURIComponent(marca)}`;
  }
  if (categoria) {
    url += `&categoria=${encodeURIComponent(categoria)}`;
  }
  if (precioMin) {
    url += `&precioMin=${encodeURIComponent(precioMin)}`;
  }
  if (precioMax) {
    url += `&precioMax=${encodeURIComponent(precioMax)}`;
  }

  fetch(url)
    .then((response) => response.json())
    .then((data) => {
      const lista = Array.isArray(data) ? data : data.content || [];
      productosGlobales = lista;

      if (!Array.isArray(data)) {
        paginaActual = data.number || 0;
        totalPaginas = data.totalPages || 1;
      }

      renderizarProductos(productosGlobales);
      actualizarControlesPaginacion();
    })
    .catch((error) => console.error("Error:", error));
}
```

**Filters supported:**
- `nombre`: Product name (text search)
- `marca`: Product brand (text search)
- `categoria`: Category ID
- `precioMin`: Minimum price
- `precioMax`: Maximum price

## Category Management

### cargarCategoriasEnFiltro()

Loads categories into the filter dropdown.

```javascript productos.js:135-159
async function cargarCategoriasEnFiltro() {
  try {
    const response = await fetch(
      "http://localhost:8080/api/categoria/listar-nombre"
    );
    if (!response.ok) throw new Error("Error al cargar categorías");

    const data = await response.json();
    const categorias = Array.isArray(data) ? data : data.content || [];

    const selectCategoria = document.getElementById("filtroCategoria");
    if (!selectCategoria) return;

    selectCategoria.innerHTML = '<option value="">Todas</option>';

    categorias.forEach((categoria) => {
      const option = document.createElement("option");
      option.value = categoria.id;
      option.textContent = categoria.nombre;
      selectCategoria.appendChild(option);
    });
  } catch (error) {
    console.error("Error al cargar categorías:", error);
  }
}
```

### cargarCategorias()

Loads categories into the product form selector with optional preselection.

<ParamField path="categoriaPreseleccionar" type="Object | string | number" default="null">
  Category to preselect (can be category object, ID, or name)
</ParamField>

```javascript productos.js:197-250
async function cargarCategorias(categoriaPreseleccionar = null) {
  try {
    const response = await fetch(
      "http://localhost:8080/api/categoria/listar-nombre"
    );
    if (!response.ok) throw new Error("Error al cargar categorías");

    const data = await response.json();
    const categorias = Array.isArray(data) ? data : data.content || [];

    const selectCategoria = document.getElementById("category");
    if (!selectCategoria) return;

    selectCategoria.innerHTML =
      '<option value="">Seleccione una categoría...</option>';

    categorias.forEach((categoria) => {
      const option = document.createElement("option");
      option.value = categoria.id;
      option.textContent = categoria.nombre;
      selectCategoria.appendChild(option);
    });

    if (categoriaPreseleccionar) {
      const categoriaId = categoriaPreseleccionar.id || categoriaPreseleccionar;
      const categoriaNombre =
        categoriaPreseleccionar.nombre || categoriaPreseleccionar;

      const opcionPorId = Array.from(selectCategoria.options).find(
        (opt) => opt.value == categoriaId
      );

      if (opcionPorId) {
        selectCategoria.value = categoriaId;
      } else {
        const opcionPorNombre = Array.from(selectCategoria.options).find(
          (opt) => opt.textContent === categoriaNombre
        );
        if (opcionPorNombre) {
          selectCategoria.value = opcionPorNombre.value;
        }
      }
    }
  } catch (error) {
    console.error("Error al cargar categorías:", error);
    alert("Error al cargar las categorías");
  }
}
```

    ```javascript productos.js:252-311
    async function registrarProducto(event) {
      event.preventDefault();

      const categoriaId = document.getElementById("category").value;

      const producto = {
        codigo: document.getElementById("code").value,
        nombre: document.getElementById("name").value,
        descripcion: document.getElementById("description").value,
        marca: document.getElementById("brand").value,
        precio: parseFloat(document.getElementById("price").value),
        categoria: parseInt(categoriaId),
      };

      try {
        const response = await fetch(
          "http://localhost:8080/api/producto/registrar",
          {
            method: "POST",
            headers: {
              "Content-Type": "application/json",
            },
            body: JSON.stringify(producto),
          }
        );
        if (!response.ok) {
          try {
            const errorData = await response.json();

            if (errorData.errores) {
              const errores = errorData.errores;
              let mensajes = [];

              if (errores.codigo) mensajes.push(`Código: ${errores.codigo}`);
              if (errores.nombre) mensajes.push(`Nombre: ${errores.nombre}`);
              if (errores.marca) mensajes.push(`Marca: ${errores.marca}`);
              if (errores.descripcion)
                mensajes.push(`Descripción: ${errores.descripcion}`);
              if (errores.precio) mensajes.push(`Precio: ${errores.precio}`);

              if (mensajes.length > 0) {
                alert(`${mensajes.join("\n")}`);
              } else {
                alert(errorData.error || "No se logró crear el producto");
              }
            } else {
              alert(errorData.error || "No se logró crear el producto");
            }
          } catch (e) {
            alert("No se logró crear el producto");
          }
          return;
        }
        alert("Producto registrado exitosamente");
        location.href = "productos.html";
      } catch (error) {
        return console.error("Error:", error);
      }
    }
    ```

    **Product Object Structure:**
    - `codigo`: Product code (string)
    - `nombre`: Product name (string)
    - `descripcion`: Description (string)
    - `marca`: Brand (string)
    - `precio`: Price (float)
    - `categoria`: Category ID (integer)

    ```javascript productos.js:170-194
      if (!confirm("¿Está seguro de eliminar el producto?")) return;
    ```

## API Endpoints

| Endpoint | Method | Description |
|----------|--------|-------------|
| `/api/producto/listar` | GET | List products with pagination, sorting, and filters |
| `/api/producto/registrar` | POST | Create new product |
| `/api/producto/actualizar/{id}` | PUT | Update existing product |
| `/api/producto/{id}` | DELETE | Delete product |
| `/api/categoria/listar-nombre` | GET | Get category list for dropdowns |

## Error Handling

The module handles validation errors from the API with field-specific messages:
- `codigo`: Code validation errors
- `nombre`: Name validation errors
- `marca`: Brand validation errors
- `descripcion`: Description validation errors
- `precio`: Price validation errors

Generic error messages are shown when specific field errors are not available.
