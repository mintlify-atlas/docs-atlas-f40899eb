## Filter Implementation

### Building Filter URL

Filters are appended to the API URL as query parameters:

```javascript
function obtenerProductos(pagina = 0) {
  let url = `http://localhost:8080/api/producto/listar?page=${pagina}&size=20&${sortHelper.getParams()}`;

  // Get filter values
  const nombre = document.getElementById("filtroNombre")?.value.trim();
  const marca = document.getElementById("filtroMarca")?.value.trim();
  const categoria = document.getElementById("filtroCategoria")?.value;
  const precioMin = document.getElementById("filtroPrecioMin")?.value;
  const precioMax = document.getElementById("filtroPrecioMax")?.value;

  // Append filters to URL if they have values
  if (nombre) {
    url += `&nombre=${encodeURIComponent(nombre)}`;
  }
  if (marca) {
    url += `&marca=${encodeURIComponent(marca)}`;
  }
  if (categoria) {
    url += `&categoria=${encodeURIComponent(categoria)}`;
  }
  if (precioMin) {
    url += `&precioMin=${encodeURIComponent(precioMin)}`;
  }
  if (precioMax) {
    url += `&precioMax=${encodeURIComponent(precioMax)}`;
  }

  fetch(url)
    .then((response) => response.json())
    .then((data) => {
      const lista = Array.isArray(data) ? data : data.content || [];
      productosGlobales = lista;

      if (!Array.isArray(data)) {
        paginaActual = data.number || 0;
        totalPaginas = data.totalPages || 1;
      }

      renderizarProductos(productosGlobales);
      actualizarControlesPaginacion();
    })
    .catch((error) => console.error("Error:", error));
}
```

## SortHelper Class

The `SortHelper` class manages sorting state and generates URL parameters.

### Class Definition

```javascript
class SortHelper {
  constructor(campoInicial = "codigo") {
    this.ordenActual = campoInicial;
    this.direccionActual = "asc";
  }

  // Changes sort order for the specified field
  ordenarPor(campo, callback) {
    // If already sorting by this field, reverse the direction
    if (this.ordenActual === campo) {
      this.direccionActual = this.direccionActual === "asc" ? "desc" : "asc";
    } else {
      // New field, sort ascending by default
      this.ordenActual = campo;
      this.direccionActual = "asc";
    }

    if (callback) callback();
  }

  // Gets sort parameters for URL
  getParams() {
    return `sort=${this.ordenActual},${this.direccionActual}`;
  }

  // Gets current field
  getCampo() {
    return this.ordenActual;
  }

  // Gets current direction
  getDireccion() {
    return this.direccionActual;
  }

  // Updates sort icons in the table
  actualizarIconos(campos) {
    campos.forEach((campo) => {
      const iconElement = document.getElementById(`icon-${campo}`);
      if (!iconElement) return;

      if (campo === this.ordenActual) {
        // Icon based on direction
        if (this.direccionActual === "asc") {
          iconElement.innerHTML = '<i class="bi bi-arrow-up"></i>';
        } else {
          iconElement.innerHTML = '<i class="bi bi-arrow-down"></i>';
        }
      } else {
        // Neutral icon
        iconElement.innerHTML = '<i class="bi bi-arrow-down-up"></i>';
      }
    });
  }
}
```

function ordenarPor(campo) {
  sortHelper.ordenarPor(campo, () => {
    sortHelper.actualizarIconos([
      "codigo",

## Loading Filter Dropdowns

Category filter dropdown is populated from the API:

```javascript
async function cargarCategoriasEnFiltro() {
  try {
    const response = await fetch(
      "http://localhost:8080/api/categoria/listar-nombre"
    );
    if (!response.ok) throw new Error("Error al cargar categorías");

    const data = await response.json();
    const categorias = Array.isArray(data) ? data : data.content || [];

    const selectCategoria = document.getElementById("filtroCategoria");
    if (!selectCategoria) return;

    selectCategoria.innerHTML = '<option value="">Todas</option>';

    categorias.forEach((categoria) => {
      const option = document.createElement("option");
      option.value = categoria.id;
      option.textContent = categoria.nombre;
      selectCategoria.appendChild(option);
    });
  } catch (error) {
    console.error("Error al cargar categorías:", error);
  }
}

cargarCategoriasEnFiltro();
```

## Complete Flow

<Steps>
  <Step title="User enters filter criteria">
    User fills in filter fields (text inputs, dropdowns, number ranges).
  </Step>
  
  <Step title="Click Buscar button">
    Triggers the `obtenerProductos(0)` or `obtenerCategorias(0)` function.
  </Step>
  
  <Step title="Build API URL">
    Function reads filter values and appends them as query parameters.
  </Step>
  
  <Step title="Include sort parameters">
    Sort parameters from `sortHelper.getParams()` are added to the URL.
  </Step>
  
  <Step title="Fetch filtered data">
    API request is sent with page, size, sort, and filter parameters.
  </Step>
  
  <Step title="Render results">
    Filtered and sorted data is displayed in the table with updated pagination.
  </Step>
</Steps>

## Sort Icons

Sort direction is indicated by Bootstrap Icons:

- `bi-arrow-down-up` - Column is not currently sorted
- `bi-arrow-up` - Column is sorted ascending
- `bi-arrow-down` - Column is sorted descending

## Key Features

- **Multiple Filter Types**: Text, select, number ranges
- **URL Parameter Based**: All filtering done server-side
- **Reusable Sort Helper**: Centralized sorting logic
- **Visual Sort Indicators**: Icons show current sort state
- **Clear Filters**: Easy reset to default state
- **Pagination Aware**: Filters reset to first page

<Note>
  Filtering and sorting are performed server-side through API query parameters. The page always resets to page 0 when filters are applied.
</Note>

<Warning>
  Make sure to use `encodeURIComponent()` when building URL parameters to properly handle special characters in filter values.
</Warning>
