## Overview

The Inmobiliaria Web frontend uses the native **Fetch API** to communicate with a RESTful backend. All API calls are asynchronous using Promises or async/await patterns with comprehensive error handling.

## API Architecture

### Base Configuration

```javascript
Base URL: http://localhost:8080
Content-Type: application/json
Methods: GET, POST, PUT, DELETE
```

### Response Format

All API responses follow a consistent paginated structure:

```json Paginated Response
{
  "content": [
    { /* item 1 */ },
    { /* item 2 */ }
  ],
  "number": 0,           // Current page (0-indexed)
  "totalPages": 5,       // Total number of pages
  "totalElements": 100,  // Total items across all pages
  "size": 20,            // Items per page
  "first": true,         // Is first page?
  "last": false          // Is last page?
}
```

### Endpoint

### Implementation

```javascript JS/productos.js (lines 7-51)
const tablaProductos = document.getElementById("tablaProductos");
let productosGlobales = [];
let paginaActual = 0;
let totalPaginas = 0;
const sortHelper = new SortHelper("codigo");

function obtenerProductos(pagina = 0) {
  let url = `http://localhost:8080/api/producto/listar?page=${pagina}&size=20&${sortHelper.getParams()}`;

  // Get filter values
  const nombre = document.getElementById("filtroNombre")?.value.trim();
  const marca = document.getElementById("filtroMarca")?.value.trim();
  const categoria = document.getElementById("filtroCategoria")?.value;
  const precioMin = document.getElementById("filtroPrecioMin")?.value;
  const precioMax = document.getElementById("filtroPrecioMax")?.value;

  // Add filters to URL if they have values
  if (nombre) url += `&nombre=${encodeURIComponent(nombre)}`;
  if (marca) url += `&marca=${encodeURIComponent(marca)}`;
  if (categoria) url += `&categoria=${encodeURIComponent(categoria)}`;
  if (precioMin) url += `&precioMin=${encodeURIComponent(precioMin)}`;
  if (precioMax) url += `&precioMax=${encodeURIComponent(precioMax)}`;

  fetch(url)
    .then((response) => response.json())
    .then((data) => {
      // API returns paginated response (object with 'content')
      const lista = Array.isArray(data) ? data : data.content || [];
      productosGlobales = lista;

      // Save pagination info
      if (!Array.isArray(data)) {
        paginaActual = data.number || 0;
        totalPaginas = data.totalPages || 1;
      }

      renderizarProductos(productosGlobales);
      actualizarControlesPaginacion();
    })
    .catch((error) => console.error("Error:", error));
}
```

### Query Parameters

| Parameter | Type | Required | Description | Example |
|-----------|------|----------|-------------|---------|
| `page` | integer | No | Page number (0-indexed) | `0` |
| `size` | integer | No | Items per page | `20` |
| `sort` | string | No | Sort field and direction | `codigo,asc` |
| `nombre` | string | No | Filter by product name | `Casa` |
| `marca` | string | No | Filter by brand | `Inmobiliaria XYZ` |
| `categoria` | integer | No | Filter by category ID | `1` |
| `precioMin` | decimal | No | Minimum price filter | `100000` |
| `precioMax` | decimal | No | Maximum price filter | `500000` |

async function registrarProducto(event) {
  event.preventDefault();

  const categoriaId = document.getElementById("category").value;

  const producto = {
    codigo: document.getElementById("code").value,
    nombre: document.getElementById("name").value,
    descripcion: document.getElementById("description").value,
    marca: document.getElementById("brand").value,
    precio: parseFloat(document.getElementById("price").value),
    categoria: parseInt(categoriaId),
  };

  try {
    const response = await fetch(
      "http://localhost:8080/api/producto/registrar",
      {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
        },
        body: JSON.stringify(producto),
      }
    );
    
    if (!response.ok) {
      try {
        const errorData = await response.json();

        // Handle validation errors with specific fields
        if (errorData.errores) {
          const errores = errorData.errores;
          let mensajes = [];

          if (errores.codigo) mensajes.push(`Código: ${errores.codigo}`);
          if (errores.nombre) mensajes.push(`Nombre: ${errores.nombre}`);
          if (errores.marca) mensajes.push(`Marca: ${errores.marca}`);
          if (errores.descripcion) mensajes.push(`Descripción: ${errores.descripcion}`);
          if (errores.precio) mensajes.push(`Precio: ${errores.precio}`);

          if (mensajes.length > 0) {
            alert(`${mensajes.join("\n")}`);
          } else {
            alert(errorData.error || "No se logró crear el producto");
          }
        } else {
          alert(errorData.error || "No se logró crear el producto");
        }
      } catch (e) {
        alert("No se logró crear el producto");
      }
      return;
    }
    
    alert("Producto registrado exitosamente");
    location.href = "productos.html";
  } catch (error) {
    return console.error("Error:", error);
  }
}

## Error Handling Patterns

### Comprehensive Error Handling

All API calls follow this pattern:

```javascript
async function apiCall() {
  try {
    const response = await fetch(url, options);
    
    if (!response.ok) {
      try {
        const errorData = await response.json();
        
        // Handle field-level validation errors
        if (errorData.errores) {
          const errores = errorData.errores;
          let mensajes = [];
          
          // Collect all field errors
          Object.keys(errores).forEach(field => {
            mensajes.push(`${field}: ${errores[field]}`);
          });
          
          alert(mensajes.join("\n"));
        } else {
          // Handle generic error message
          alert(errorData.error || "Error en la operación");
        }
      } catch (e) {
        // If error response is not JSON
        alert("Error en la operación");
      }
      return;
    }
    
    // Success handling
    const data = await response.json();
    // Process data...
    
  } catch (error) {
    // Network or other errors
    console.error("Error:", error);
    alert("Error de conexión");
  }
}
```

## Common Integration Issues

<AccordionGroup>
<Accordion title="CORS Errors" icon="shield-exclamation">

**Symptom:** Browser console shows CORS policy error

**Solution:** Configure backend to allow frontend origin:

```java Backend Configuration (Spring Boot)
@Configuration
public class CorsConfig {
    @Bean
    public WebMvcConfigurer corsConfigurer() {
        return new WebMvcConfigurer() {
            @Override
            public void addCorsMappings(CorsRegistry registry) {
                registry.addMapping("/api/**")
                    .allowedOrigins("http://localhost:5500")
                    .allowedMethods("GET", "POST", "PUT", "DELETE", "OPTIONS")
                    .allowedHeaders("*");
            }
        };
    }
}
```

</Accordion>

<Accordion title="404 Not Found" icon="magnifying-glass-minus">

**Symptom:** API returns 404 for valid endpoints

**Possible Causes:**
1. Backend not running on `http://localhost:8080`
2. Incorrect endpoint path in JavaScript
3. Backend context path is different

**Solution:**
- Verify backend is running: `curl http://localhost:8080/api/producto/listar`
- Check backend logs for actual endpoint paths
- Update frontend configuration if backend URL differs

</Accordion>

<Accordion title="Empty Response" icon="circle-question">

**Symptom:** API call succeeds but returns empty array

**Debugging:**
```javascript
fetch(url)
  .then(response => {
    console.log('Status:', response.status);
    return response.json();
  })
  .then(data => {
    console.log('Response:', data);
    console.log('Is Array:', Array.isArray(data));
    console.log('Has content:', data.content);
  });
```

**Common Issues:**
- Database is empty (seed some data)
- Pagination parameters are incorrect
- Filters are too restrictive

</Accordion>

<Accordion title="Network Errors" icon="wifi-slash">

**Symptom:** "Failed to fetch" or "Network error"

**Checklist:**
- [ ] Backend server is running
- [ ] Backend is accessible at configured URL
- [ ] No firewall blocking requests
- [ ] HTTPS/HTTP protocol matches between frontend and backend

**Test Backend Manually:**
```bash
curl -v http://localhost:8080/api/producto/listar
```

</Accordion>
</AccordionGroup>
